添加一元的加法和减法# 

expr    ::= mul ("+" mul | "-" mul)*
mul     ::= unary ("\*" unary | "/" unary)*
unary   ::= ("+" | "-") unary
          | primary
primary ::= "(" expr ")" | NUM
